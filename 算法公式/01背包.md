/**
 * let i = 0 个数
 * let j = 0 重量
 * 不拿第i个物品,最大价值 f[i][j] = f[i-1][j]
 * 拿第i个物品，最大价值  f[i][j] = f[i-1][j-v[i]] + w[i]
 */
 function maxValue(N,V,v,w){
     let i,j
     let f = []
     f[0] = [0,0,0,0,0,0]
     f[0][0] = 0
     for(i = 1; i< N; i++){
         for(j = V; j > 0; j--){
             f[i] = f[i] || [0,0,0,0,0,0]
             f[i][j] = f[i-1][j]
             if(j>=v[i]){
                 f[i][j] = Math.max(f[i][j], f[i-1][j-v[i]]+w[i])
             }
         }
     }
     console.log(f[N-1][V])
 }
//  maxValue(4,5,[0,1,2,3,4],[0,2,4,4,5])
